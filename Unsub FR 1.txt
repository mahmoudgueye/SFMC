%%[
/*set @longid = "0030Y00000bbTuJQAU"*/

set @longid = _subscriberkey
  
/*Make sure all characters of subscriberKey are not in lower case. If they are, it means the email has been sent via Audience Builder and the subKey with correct case needs to be retrieved*/  
IF Length(RegExMatch(@longid, '[A-Z]', 0)) == 0 THEN
SET @longid =Lookup("Clients", "Id" , "Id", @longid)
ENDIF
  
var @Contact, @Email,@PersonHasOptedInOfEmail,@KeringId,@ContactRow,@AccountId,@SalesforceObject,@DataExtensionToUpdate

SET @sid = SubscriberID
SET @jid = JobID
SET @batchid = _JobSubscriberBatchID

set @Contact = RetrieveSalesforceObjects("Account","Id,PersonEmail,PersonHasOptedInOfEmail__c,KeringId__c","PersonContactId", "=", @longid) 
set @Lead = RetrieveSalesforceObjects("Lead","Id,Email,PersonHasOptedInOfEmail__c,KeringId__c","Id", "=", @longid) 

var @status
set @status =Lookup("ENT._Subscribers", "Status" , "SubscriberKey", _subscriberkey)

if (RowCount(@Contact) == 1) THEN
Set @DataExtensionToUpdate = 'Clients'
Set @SalesforceObject = 'Account'
Set @ContactRow = ROW(@Contact,1) 
Set @Id = FIELD(@ContactRow,"Id")
Set @Email = FIELD(@ContactRow,"PersonEmail")
Set @PersonHasOptedInOfEmail = FIELD(@ContactRow,"PersonHasOptedInOfEmail__c")
Set @KeringId= FIELD(@ContactRow,"KeringId__c")

   
elseif (RowCount(@Lead) == 1) THEN
Set @DataExtensionToUpdate = 'NonQualifiedProspects'
Set @SalesforceObject = 'Lead'
Set @LeadRow = ROW(@Lead,1) 
Set @Id = FIELD(@LeadRow,"Id")
Set @Email = FIELD(@LeadRow,"Email")
Set @PersonHasOptedInOfEmail = FIELD(@LeadRow,"PersonHasOptedInOfEmail__c")
Set @KeringId= FIELD(@LeadRow,"KeringId__c")
    
ELSE 
Redirect("https://www.bottegaveneta.com/")
ENDIF
